A Trie implementation using a generic HashDB backing database.

Use it as a Trie trait object. You can use db() to get the backing database object, keys to get the keys belonging to the trie in the backing database, and db_items_remaining() to get which items in the backing database do not belong to this trie. If this is the only trie in the backing database, then db_items_remaining() should be empty.

Example

extern crate ethcore_util as util;
use util::trie::*;
use util::hashdb::*;
use util::memorydb::*;
use util::hash::*;
use util::rlp::*;

fn main() {
  let mut memdb = MemoryDB::new();
  let mut root = H256::new();
  TrieDBMut::new(&mut memdb, &mut root).insert(b"foo", b"bar");
  let t = TrieDB::new(&memdb, &root);
  assert!(t.contains(b"foo"));
  assert_eq!(t.get(b"foo").unwrap(), b"bar");
  assert!(t.db_items_remaining().is_empty());
}
